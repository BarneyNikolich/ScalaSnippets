Using the Any type. Match statement handles Strings or Ints or Chars 


 		def matchingScalaAnyType(something: Any): String = something match 
 		{
    			case n: Int => "Number entered " + n
    			case s: String => "String entered " + s
    			case c: Char => "Char entered " + c  
    			case _ => "Something else"
  		}

Simple ParseInt method with Exception Handling Try Catch- takes in String and converts to Int. 


		 def parseInt(s: String): Int = try 
		{
    		s.toInt
  		} catch {
    		case e: NumberFormatException => -1
  		}

		def convertStringToInt() 
		{
   			println("Enter the String to convert: ")
    		val input = readLine()
   		 	println(parseInt(input))
  		}

Example of Class 



		class Point(xP: Int, yP: Int) 
		{
   		 //    Create local variables from the parameters
  		 	var x: Int = xP
  			var y: Int = yP

   		 //    Define a method in the class
   		 def move(dx: Int, dy: Int): Unit = 
   		 {
     		 x = x + dx
    		 y = y + dy
    	  	}

   		 //Must override toString method as it is a default method.
   		 override def toString(): String = "(" + x + ", " + y + ")"
  		}

  		//  Instantiate an instance of a class.
  		val pt = new Point(5, 5)
  		println(pt.toString())
  		pt.move(10, 10)
  		println(pt.toString())
